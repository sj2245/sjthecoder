import { blogPosts } from '../../server';

export async function getStaticPaths() {
  const paths = blogPosts.map(post => ({
    params: { id: post.id.toString() },
  }));
  return { paths, fallback: false };
}

export async function getStaticProps({ params }) {
  const post = blogPosts.find(p => p.id.toString() === params.id);
  if (!post) {
    return {
      notFound: true,
    };
  }
  return { props: { post } };
}

<div style={{textAlign: `center`}}>
  # Post
</div>

export default function PostPage({ post }) {
  return <>
    <div id={`blog-post-${post.id}`} className="blogPost">
      <div className="blogContent project">
        <img alt={`Post Image`} className="ProjImg" src={post.image} />
        <div className="projDes">
          <h3>{post.title}</h3>
          <div className={`postText`} dangerouslySetInnerHTML={{ __html: post.text }} />
        </div>
      </div>
    </div>
  </>
}
